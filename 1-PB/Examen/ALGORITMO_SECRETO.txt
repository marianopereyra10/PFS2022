function cargar(v) {
    for (let i = 0; i < v.length; i++)
            v[i]=Math.floor(Math.random()*100);
}
function mostrar(v,n) {
    let c="";

            c+= v[i] + " ";
    console.log(c);
}
function funcion(v,n) {
